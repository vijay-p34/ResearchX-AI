import type { Principal } from "@icp-sdk/core/principal";
export interface Some<T> {
    __kind__: "Some";
    value: T;
}
export interface None {
    __kind__: "None";
}
export type Option<T> = Some<T> | None;
export interface Paper {
    id: string;
    url: string;
    status: PaperStatus;
    title: string;
    collectionId?: string;
    journal: string;
    createdAt: bigint;
    tags: Array<string>;
    year: bigint;
    summary: string;
    authors: Array<string>;
    notes: string;
    abstract: string;
}
export enum PaperStatus {
    Done = "Done",
    Unread = "Unread",
    Reading = "Reading"
}
export interface backendInterface {
    addCollection(id: string, name: string, description: string): Promise<void>;
    addPaper(id: string, title: string, authors: Array<string>, abstract: string, year: bigint, journal: string, url: string, tags: Array<string>): Promise<void>;
    addQAEntry(id: string, paperId: string, question: string, answer: string): Promise<void>;
    getPaper(id: string): Promise<Paper>;
    getPapersByTag(tag: string): Promise<Array<Paper>>;
    searchPapers(keyword: string): Promise<Array<Paper>>;
    updatePaper(id: string, title: string, authors: Array<string>, abstract: string, year: bigint, journal: string, url: string, tags: Array<string>): Promise<void>;
}
